building_hydroponics_farm = {
	base_buildtime = @b1_time
	category = resource

	building_sets = {
		hydroponics
	}

	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_food = yes
				is_ai = no
			}
			is_wilderness_empire = no
		}
		NOT = { is_planet_class = pc_machine }
		NOT = { has_modifier = resort_colony }
		OR = {
			has_carrier_flag = ignore_ai_building_limitations
			owner = { is_ai = no }
			# AI is allowed to build up to two of these.
			num_buildings = { type = building_hydroponics_farm value < 2 }
		}
	}
	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				AND = {
					is_ai = yes
					country_uses_food = no
				}
				is_wilderness_empire = yes
			}
		}
	}

	inline_script = {
		script = jobs/farmers_add
		AMOUNT = @building_static_jobs
	}

	resources = {
		category = planet_buildings
		inline_script = {
			script = buildings/nomadic_cost_switcher
			REGULAR_RESOURCE = minerals
			NOMADIC_RESOURCE = volatile_motes
			COST = @b1_minerals
			NOMADIC_COST_MULT = @nomadic_minerals_to_strategic_ratio
		}
		upkeep = {
			energy = 2
		}
	}
	prerequisites = {
		"tech_hydroponics"
	}
	
	ai_weight_coefficient = @static_jobs_ai_weight_coefficient

	# EUTAB
	upgrades = {
		building_eutab_hydroponics_tower # EUTAB
	}
}
